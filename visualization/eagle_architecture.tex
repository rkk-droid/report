\begin{document}
\begin{tikzpicture}

% ── Input nodes ───────────────────────────────────────────────────────────────
\node[innode, fill=nodeblue!25, draw=nodeblue] (v) at (0,0) {$v$};
\node[below=0.1cm of v, font=\scriptsize, color=nodeblue!80!black]
    {$\mathbf{x}_v(t)$};

\node[innode, fill=nodeorange!25, draw=nodeorange] (u) at (22,0) {$u$};
\node[below=0.1cm of u, font=\scriptsize, color=nodeorange!80!black]
    {$\mathbf{x}_u(t)$};

% ── Time-aware module ─────────────────────────────────────────────────────────
\node[sblk, fill=taCyanLight, draw=taCyan] (rv) at (3.5, 5.5)
    {Recent Nbrs\\$N_v(t,k_r)$\\{\tiny feat: $\mathbf{x}_{v_i}, \mathbf{e}_{v,v_i}$}};

\node[sblk, fill=taCyanLight!80, draw=taCyan] (hv_ta) at (7.5, 5.5)
    {Avg.~concat\\$\mathbf{h}_v^{\mathrm{ta}}(t)$};

\node[mblk, fill=taCyanLight!60, draw=taCyan] (ta_mlp) at (11, 5.5)
    {2-layer MLP $+\;\sigma$\\
     {\tiny $\mathbf{W}_1[\mathbf{h}_v^{\mathrm{ta}},\mathbf{h}_u^{\mathrm{ta}}]
       \!\to\!\mathrm{ReLU}\!\to\!\mathbf{W}_2\!\to\!\sigma$}};

\node[sblk, fill=taCyanLight!80, draw=taCyan] (hu_ta) at (14.5, 5.5)
    {Avg.~concat\\$\mathbf{h}_u^{\mathrm{ta}}(t)$};

\node[sblk, fill=taCyanLight, draw=taCyan] (ru) at (18.5, 5.5)
    {Recent Nbrs\\$N_u(t,k_r)$\\{\tiny feat: $\mathbf{x}_{u_i}, \mathbf{e}_{u,u_i}$}};

\node[sc, fill=taCyan!35, draw=taCyan!80!black] (s_ta) at (11, 3.5)
    {$s_{v,u}^{\mathrm{ta}}$};

% Module bounding box (time-aware)
\begin{scope}[on background layer]
\node[modbox, fill=taCyan!6, draw=taCyan!70!black,
      fit=(rv)(ru)(hv_ta)(hu_ta)(ta_mlp)(s_ta),
      label={[font=\small\bfseries, color=taCyan!80!black, yshift=2pt]above:%
             Time-aware Module}] (ta_mod) {};
\end{scope}

% ── Structure-aware module ────────────────────────────────────────────────────
\node[sblk, fill=saGreenLight, draw=saGreen] (pv) at (3.5, -5.5)
    {Top-$k_s$ T-PPR\\$N_v(\bm{\pi}_v,k_s)$\\{\tiny temporal PPR}};

\node[sblk, fill=saGreenLight!80, draw=saGreen] (piv) at (7.5, -5.5)
    {T-PPR vector\\$\bm{\pi}_v(t)$};

\node[mblk, fill=saGreenLight!60, draw=saGreen] (sa_dot) at (11, -5.5)
    {Intersection $\cap$ + dot-product\\
     {\tiny $\displaystyle\sum_{v_i\in N_v\cap N_u}
       \bm{\pi}_v[v_i]\cdot\bm{\pi}_u[v_i]$}\\
     {\tiny (no trainable parameters)}};

\node[sblk, fill=saGreenLight!80, draw=saGreen] (piu) at (14.5, -5.5)
    {T-PPR vector\\$\bm{\pi}_u(t)$};

\node[sblk, fill=saGreenLight, draw=saGreen] (pu) at (18.5, -5.5)
    {Top-$k_s$ T-PPR\\$N_u(\bm{\pi}_u,k_s)$\\{\tiny temporal PPR}};

\node[sc, fill=saGreen!35, draw=saGreen!80!black] (s_sa) at (11, -3.5)
    {$s_{v,u}^{\mathrm{sa}}$};

% Module bounding box (structure-aware)
\begin{scope}[on background layer]
\node[modbox, fill=saGreen!6, draw=saGreen!70!black,
      fit=(pv)(pu)(piv)(piu)(sa_dot)(s_sa),
      label={[font=\small\bfseries, color=saGreen!80!black, yshift=-2pt]below:%
             Structure-aware Module \normalfont(T-PPR, no training)}]
    (sa_mod) {};
\end{scope}

% ── Hybrid module ─────────────────────────────────────────────────────────────
\node[hblk, fill=hybridLight, draw=hybridGold] (hybrid) at (11, 0)
    {\textbf{Adaptive Hybrid Fusion}\\[4pt]
     $s_{v,u}^{\mathrm{hy}}(t)=
       \lambda\!\cdot\!\bigl(e^{-\bar{t}_v}+e^{-\bar{t}_u}\bigr)
       \cdot s_{v,u}^{\mathrm{ta}}(t)+s_{v,u}^{\mathrm{sa}}(t)$\\[3pt]
     {\scriptsize $\bar{t}_v$ — avg.\ time gap of $k_r$ recent neighbors of $v$}};

% ── Output (строго под hybrid, ниже всего diagram) ───────────────────────────
\node[oblk, fill=outLight, draw=outRed] (output) at (11, -9.2)
    {\textbf{Link Score}\\$\hat{y}_{v,u}(t)$};

% ── T-PPR update note (нижний левый угол, вне всех модулей) ──────────────────
\node[draw, rounded corners=4pt, fill=gray!8, line width=0.8pt,
      text width=5.2cm, align=center, font=\scriptsize]
    (tppr_note) at (3.5, -8.8)
    {T-PPR update:\\
     $\pi_v^{l+1}(t)=\alpha\mathbf{P}(t)\pi_v^l(t)+(1-\alpha)\mathbf{r}_v(t)$};

% ═════════════════════════════════════════════════════════════════════════════

% ── v / u  →  Recent neighbors ───────────────────────────────────────────────

    node[la, pos=0.55, above left]{select $k_r$} (rv.south west);
\draw[tarr] (u.north) to[out=100, in=-30]
    node[la, pos=0.55, above right]{select $k_r$} (ru.south east);

% ── Recent neighbors  →  Avg concat ──────────────────────────────────────────
\draw[arr] (rv.east)    -- node[la, above]{avg concat} (hv_ta.west);
\draw[arr] (ru.west)    -- node[la, above]{avg concat} (hu_ta.east);

% ── Avg concat  →  MLP ───────────────────────────────────────────────────────
\draw[arr] (hv_ta.east) -- (ta_mlp.west);
\draw[arr] (hu_ta.west) -- (ta_mlp.east);

% ── MLP  →  s_ta ─────────────────────────────────────────────────────────────
\draw[tarr] (ta_mlp.south) -- (s_ta.north);

% ── s_ta  →  Hybrid ──────────────────────────────────────────────────────────
\draw[tarr] (s_ta.south)
    -- node[la, right, pos=0.5]{$s_{v,u}^{\mathrm{ta}}$}
    (hybrid.north);

% ── v / u  →  T-PPR nodes ────────────────────────────────────────────────────
\draw[sarr] (v.south) to[out=-80, in=150]
    node[la, pos=0.55, below left]{T-PPR $k_s$} (pv.north west);
\draw[sarr] (u.south) to[out=-100, in=30]
    node[la, pos=0.55, below right]{T-PPR $k_s$} (pu.north east);

% ── T-PPR nodes  →  pi vectors ───────────────────────────────────────────────
\draw[arr] (pv.east)    -- node[la, below]{scores} (piv.west);
\draw[arr] (pu.west)    -- node[la, below]{scores} (piu.east);

% ── pi vectors  →  dot-product ───────────────────────────────────────────────
\draw[arr] (piv.east)   -- (sa_dot.west);
\draw[arr] (piu.west)   -- (sa_dot.east);

% ── dot-product  →  s_sa ─────────────────────────────────────────────────────
\draw[sarr] (sa_dot.north) -- (s_sa.south);

% ── s_sa  →  Hybrid ──────────────────────────────────────────────────────────
\draw[sarr] (s_sa.north)
    -- node[la, right, pos=0.5]{$s_{v,u}^{\mathrm{sa}}$}
    (hybrid.south);

% ── Adaptive weight arrows (v / u  →  Hybrid, dashed orange) ─────────────────
% v → hybrid.west:  goes straight at y=0, below time-aware module
\draw[darr] (v.east)
    to[out=0, in=180]
    node[la, below, pos=0.5]{$\bar{t}_v=\tfrac{1}{k_r}\sum\Delta\tau$}
    (hybrid.west);

% u → hybrid.east:  mirrors, goes below structure-aware path
\draw[darr] (u.west)
    to[out=180, in=0]
    node[la, below, pos=0.5]{$\bar{t}_u$}
    (hybrid.east);

% ── Hybrid  →  Output (прямая вертикальная стрелка) ─────────────────────────
\draw[-{Stealth[length=9pt,width=6pt]}, line width=2pt, color=outRed]
    (hybrid.south) -- (output.north);
